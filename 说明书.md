public static JsonNode transform(JsonNode input) {
        JsonNode result = input.deepCopy();
        processNode(result);
        return result;
    }


    private static void processNode(JsonNode node) {
        if (!node.isObject()) return;
        ObjectNode obj = (ObjectNode) node;

        if (isPrimitiveTypeDef(obj)) {
            wrapNode(obj);
            return;
        }

        Iterator<Map.Entry<String, JsonNode>> fields = obj.fields();
        while (fields.hasNext()) {
            Map.Entry<String, JsonNode> entry = fields.next();
            JsonNode child = entry.getValue();
            if (child.isObject() || child.isArray()) {
                processNode(child);
            }
        }
    }


    private static boolean isPrimitiveTypeDef(ObjectNode obj) {
        JsonNode type = obj.get("type");
        if (type == null || !type.isTextual()) return false;
        String typeStr = type.asText();
        if (typeStr.equals("object") || typeStr.equals("array")) return false;
        if (obj.has("properties") || obj.has("items")) return false;
        return true;
    }


    private static void wrapNode(ObjectNode obj) {
        ObjectNode wrapper = MAPPER.createObjectNode();
        wrapper.put("type", "object");
        ObjectNode props = wrapper.putObject("properties");
        props.set("value", obj.deepCopy());
        props.set("pageIndex",  MAPPER.createObjectNode().put("type", "number"));
        props.set("lineIndex",  MAPPER.createObjectNode().put("type", "number"));
        props.set("blockIndex", MAPPER.createObjectNode().put("type", "number"));
        obj.removeAll();
        obj.setAll(wrapper);
    }
